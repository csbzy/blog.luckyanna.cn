2015-10-27

# Pagination
## Set per_page to 0 to disable pagination
per_page: 6
pagination_dir: page
archive_generator:
  yearly: true
  monthly: true
  per_page: 0
category_generator:
  per_page: 0
tag_generator:
  per_page: 0
