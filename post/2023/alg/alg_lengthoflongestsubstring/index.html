func longestsubstring(s string) int {
	if len(s) < 2 {
		return len(s)
	}
	var (
		left      = 0
		right     = 0
		maxLength = 0
		exists    = map[byte]int{}
	)

	for right < len(s) {
		if idx, ok := exists[s[right]]; ok {
			left = maxIn(left, idx+1)
		}

		exists[s[right]] = right
		if right-left+1 > maxLength {
			maxLength = right - left + 1
		}
		right++

	}
	return maxLength
}
